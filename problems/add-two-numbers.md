```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */

var addTwoNumbers = function(l1, l2) {
    var fakeHead = new ListNode(-1),
        cur = fakeHead;
    var sum = 0, 
        append = false;
    while (l1 || l2 || append) {
        if (append) {
            sum = (l1&&l1.val) + (l2&&l2.val) + 1;
        } else {
             sum = (l1&&l1.val) + (l2&&l2.val)
        }
        if (sum>=10) {
            append = true;
            cur.next = new ListNode(sum%10);
        } else {
            append = false;
            cur.next = new ListNode(sum);
        }
        l1 = l1&&l1.next;
        l2 = l2&&l2.next;
        cur = cur.next;
    }
    return fakeHead.next;
};

```

## 2 两数相加

记住要写一个进位器，然后因为进位器是从个位开始的，所以就不用倒过来了。如果用JavaScript内置的数组倒序，那么会需要很长时间，且大数会出现运算精度损失的问题。

可以使用fakeHead作为开始，fakeHead下一个节点作为返回的链表的第一个节点。
