```typescript
/**
 Do not return anything, modify matrix in-place instead.
 */
function rotate(matrix: number[][]): void {
  const n = matrix.length;

  for (let i = 0; i < n / 2; i++) {
    [matrix[n - i - 1], matrix[i]] = [matrix[i], matrix[n - i - 1]];
  }

  for (let i = 0; i < n; i++) {
    for (let j = 0; j < i; j++) {
      [matrix[i][j], matrix[j][i]] = [matrix[j][i], matrix[i][j]];
    }
  }
};

```

# 解法一：使用额外的矩阵进行旋转（不符合题意）：
```typescript
/**
 Do not return anything, modify matrix in-place instead.
 */
function rotate(matrix: number[][]): void {
    const n = matrix.length;
    const mem: number[][] = new Array(n).fill([]).map(i => new Array(n).fill(-Infinity));

    for (let i = 0; i < n; i++) {
        for (let j = 0; j < n; j++) {
            mem[j][n - i - 1] = matrix[j][n - i - 1];
            if (mem[i][j] === -Infinity) {
                matrix[j][n - i - 1] = matrix[i][j];
            } else {
                matrix[j][n - i - 1] = mem[i][j];
            }

        }
    }
};
```

# 解法二： 使用翻转法
先把第一行写到最后一行，第二行写到倒数第二行....

然后这个新的方阵沿对角线变为它的对称方阵即可。

```typescript
/**
 Do not return anything, modify matrix in-place instead.
 */
function rotate(matrix: number[][]): void {
  const n = matrix.length;

  for (let i = 0; i < n / 2; i++) {
    [matrix[n - i - 1], matrix[i]] = [matrix[i], matrix[n - i - 1]];
  }

  for (let i = 0; i < n; i++) {
    for (let j = 0; j < i; j++) {
      [matrix[i][j], matrix[j][i]] = [matrix[j][i], matrix[i][j]];
    }
  }
};
```
